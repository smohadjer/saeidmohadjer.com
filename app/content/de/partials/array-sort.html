Sorting an array by using another array to define the sort order

javascript

Sometimes you need to display data that you receive from backend in a custom order in frontend. Following snippet shows how to order objects in an array that you receive from backend via an order defined in a second array. Note that to avoid mutation we muss use WEB API's structuredClone() method instead of spread syntax (...) as spread syntax only does a shallow copy of array items.

const myArray = [
  {name: 'firstname', value: 'Tom'},
  {name: 'age', value: 61},
  {name: 'lastname', value: 'Cruise'}
];
const mySortOrder = ['firstname', 'lastname', 'age'];
const sortArray = (myArray, mySortOrder) => {
  myArray.sort((a, b) => {
    const indexA = mySortOrder.indexOf(a.name);
    const indexB = mySortOrder.indexOf(b.name);
    return indexA - indexB;
  });
  return myArray;
}
const sortedArray = sortArray(structuredClone(myArray), mySortOrder);
console.log(sortedArray);
